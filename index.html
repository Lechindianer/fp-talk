Object-Oriented


class PasswordTester {
  private password: Password;
  private email: Email;

  constructor(password: Password, email: Email) {
    this.password = password;
    this.email = email;
  }

  public setPassword(password: Password) {
    this.password = password;
  }

  public setEmail(email: Email) {
    this.email = email;
  }

  public isInvalidPassword(): boolean {
    return !this.password.contains(this.email);
  }
}
